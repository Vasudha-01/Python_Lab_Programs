<!DOCTYPE html>

<html lang="en" ng-app="myApp">

<head>

<meta charset="UTF-8">

<title>AngularJS Directives - Countries List</title>

<script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.2/angular.min.js"></script>

</head>

<body>

<div ng-controller="myController">

<h1>Countries List with Continents</h1>

<country-list countries="countryData"></country-list>

</div>

<script>

angular.module('myApp', [])

.controller('myController', function($scope) {

$scope.countryData = [

{ name: 'USA', continent: 'North America' },

{ name: 'Canada', continent: 'North America' },

{ name: 'Brazil', continent: 'South America' },

{ name: 'France', continent: 'Europe' },

{ name: 'Germany', continent: 'Europe' },

{ name: 'Australia', continent: 'Oceania' },

{ name: 'China', continent: 'Asia' },

{ name: 'South Africa', continent: 'Africa' }

];

})
.directive('countryList', function() {

return {

restrict: 'E',

scope: {

countries: '='

},

template: '<ul>' +
            '<li ng-repeat="country in countries">' +
              '<strong>{{ country.name}}</strong> {{ country.continent }}' +
            '</li>' +
          '</ul>'
};

});

</script>

</body>

</html>

<!DOCTYPE html>
<html lang="en" ng-app="myApp">
<head>
    <meta charset="UTF-8">
    <title>AngularJS Button Interactions</title>
    <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.2/angular.min.js"></script>
</head>
<body ng-controller="myController">
    <button ng-click="incrementCount()" ng-disabled="disabled">Click Me!</button>
    <button ng-show="showButton" ng-click="toggleShow()">Hide Button to Click Me!</button>
    <button ng-hide="showButton" ng-click="toggleShow()">Show Button</button>
    <p>Total times the button is clicked: {{ clickCount }}</p>

    <script>
        angular.module('myApp', [])
        .controller('myController', function($scope) {
            $scope.clickCount = 0;
            $scope.disabled = false;
            $scope.showButton = true;

            $scope.incrementCount = function() {
                $scope.clickCount++;
            };

            $scope.toggleShow = function() {
                $scope.showButton = !$scope.showButton;
            };
        });
    </script>
</body>
</html>
